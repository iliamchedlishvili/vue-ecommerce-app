
# Simple Flow Without Branches
git add .                   # this command stages files to commit
git status                  # check status modified files 
git commit -m 'comment'     # saves only staged changes to repository locally
git push                    # send commits to the remote repository
git fetch                   # Downloads changes only
git pull                    # Downloads + merges

# With Branches

# 1) Start from latest main
git checkout main                     # switch to main
git pull                              # get latest changes from remote

# 2) Create feature branch
git checkout -b feature/my-new-branch # create & switch to feature branch
git checkout feature/my-new-branch  # if already exists

# 3) Do your work, then commit
git add .
git commit -m "comment"

# 4) Push feature branch (first time)
git push -u origin feature/my-new-branch
# Next times you can just: git push

# 5) Go back to main and update it
git checkout main
git pull                              # important: main may have new changes

# 6) Merge feature into main (locally)
git merge feature/my-new-branch

# 7) Push updated main to remote
git push

git branch -d feature/my-new-branch          # delete local branch
git push origin --delete feature/my-new-branch   # delete remote branch
