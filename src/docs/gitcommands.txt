// Git commands flow

git add .                   // this command stages files to commit
git status                  // check status modified files 
git commit -m 'comment'     // saves only staged changes to repository locally
git push                    // send commits to the remote repository
git fetch                   // Downloads changes only
git pull                    // Downloads + merges

// branches

//Feature branch work
git checkout -b feature/my-new-branch       //* create new branch
git checkout feature/my-new-branch          // if branch already exists
git add .                                   //*
git commit -m "Your commit message"         //*
git push -u origin feature/my-new-branch    //* push branch to master

// Merge into master
git checkout master                         //* Switch to master (or main) before merging
git pull                                    //* Pull the latest changes to avoid conflicts:
git merge feature/my-new-branch             //* Merge your branch into master

// if conflict occurs Manually fix the conflicts in files and then run this commands 
git add .                                   // this steps is required if there is a conflict
git commit -m 'comment'                     // this steps is required if there is a conflict

// last push changes to master
git push                                    //* send commits to the remote repository
